6.1 . Peterson's Algorithm - The First Correct Solution:

// Process P0
// Indicate that Process P0 wants to enter the critical section
Flag[0] = True;
// Give priority to process P1 in case both want to enter
Turn = j;
// Busy wait while P1 is interested/ready and it is P1’s turn
while (Flag[1] == True && Turn == j);

// CRITICAL SECTION

// Indicate that Process P0 has exited the critical section
Flag[0] = False;
// Process P1
// Indicate that Process P1 wants to enter the critical section
Flag[1] = True;
// Give priority to process P0 in case both want to enter
Turn = i;
// Busy wait while P0 is interested/ready and it is P0’s turn
while (Flag[0] == True && Turn == i);

// CRITICAL SECTION

// Indicate that Process P1 has exited the critical section
Flag[1] = False;
